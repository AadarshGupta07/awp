Aim: Given the code as sequence of characters and given the probability of characters, Write a program to calculate average length of each code.

    
        #include< stdio.h>
        #include< conio.h>
        #include< string.h>
        
        void main()
        {
            char letter[5];
            float p[5], avg_len = 0.0;
            char c[5][100];
            int i;
        
            clrscr();
            printf("Enter letter:");
            for (i = 0; i < 5; i++)
            {
                scanf("\n%c", &letter[i]);
            }
        
            printf("Enter probability:\n");
            for (i = 0; i < 5; i++)
            {
                scanf("%f", &p[i]);
            }
        
            printf("Enter codeword value\n");
            for (i = 0; i < 5; i++)
            {
                scanf("%s", c[i]);
            }
        
            for (i = 0; i < 5; i++)
            {
                avg_len = avg_len + (p[i] * strlen(c[i]));
            }
        
            printf("Average is %f", avg_len);
            getch();
        }
