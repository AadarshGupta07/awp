Aim: Implement the various low pass and high pass filtering mechanisms.

    
        % Low Pass Filter
        img = imread('image.jpg');

        % Convert the image to grayscale
        img_gray = rgb2gray(img);

        % Define the low-pass filter kernel
        kernel = ones(3, 3) / 9;

        % Apply the low-pass filter to the color image
        filtered_color = imfilter(img, kernel);

        % Apply the low-pass filter to the grayscale image
        filtered_gray = imfilter(img_gray, kernel);

        % Display the original and filtered images
        figure;
        subplot(2, 2, 1);
        imshow(img);
        title('Original Color Image');
        subplot(2, 2, 2);
        imshow(filtered_color);
        title('Low-Pass Filtered Color Image');
        subplot(2, 2, 3);
        imshow(img_gray);
        title('Original Grayscale Image');
        subplot(2, 2, 4);
        imshow(filtered_gray);
        title('Low-Pass Filtered Grayscale Image');

        
        % High Pass Filter
        img = imread('image.jpg');

        % Convert the image to grayscale
        img_gray = rgb2gray(img);

        % Define the high-pass filter kernel
        kernel = [-1 -1 -1; -1 8 -1; -1 -1 -1];

        % Apply the high-pass filter to the color image
        filtered_color = imfilter(img, kernel);

        % Apply the high-pass filter to the grayscale image
        filtered_gray = imfilter(img_gray, kernel);

        % Display the original and filtered images
        figure;
        subplot(2, 2, 1);

        title('Original Color Image');
        subplot(2, 2, 2);

        imshow(img_gray);
