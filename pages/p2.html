      % (a) Contrast Adjustment
      i =imread('image.jpg'); 
      
      % Increase Contrast
      a= i*2; 
      
      % Decrease Contrast
      b=i*0.5;   
      
      % Display of image, Subplot with 1 row and 3 columns 
      subplot(1,3,1),imshow(i),title('Original Image');
      subplot(1,3,2),imshow(a),title('Increased in Contrast');
      subplot(1,3,3),imshow(b),title('Decreased in Contrast');

      % (b) Histogram Equalization
      % Reading Image
      a =imread('image.jpg'); 

      % Perform Histogram Equalization on Image Perform histogram equalization on image
      heq = histeq(a); 

      % Display the Original & Equalized Images side by side
      subplot (221); imshow(a); title('Original Image');
      subplot (222); imshow(heq); title('Equalized Image'); 

      % Display the Histogram of Original & Equalized Images side by side
      subplot (223); imhist(a); title('Histogram of Original Image');
      subplot (224); imhist(heq); title('Histogram of Equalized Image');
