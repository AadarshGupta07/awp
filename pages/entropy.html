      // Pract 1 a: Entropy Calculation
      #include< stdio.h>

      void main()
      {
          float p[5], Info[5], I;
          int i, j;
          clrscr();
          printf("Enter the probability of 5 characters:");
          for (i = 0; i < 5; i++)
          {
              scanf("%f", &p[i]);
          }
          for (j = 0; j < 5; j++)
          {
              Info[j] = p[j] * (log(1 / p[j]));
          }
          I = Info[0] + Info[1] + Info[2] + Info[3] + Info[4];
          printf("The entropy of this probability is: %f", I);
          getch();
      
          // Pract 1 b: Occurrences Calculation
          int count;
          char s[100];
          clrscr();
          printf("Enter a string:");
          gets(s);
          int n = strlen(s);
          for (i = 0; i < n; i++)
          {
              count = 1;
              if (s[i])
              {
                  for (j = i + 1; j < n; j++)
                  {
                      if (s[i] == s[j])
                      {
                          count++;
                          s[j] = '\0';
                      }
                  }
                  printf("'%c' -> %d\n", s[i], count);
              }
          }
          getch();
      }
