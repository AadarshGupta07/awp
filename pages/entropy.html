      // Pract 1 a: Entropy Calculation
        #include < stdio.h>
        #include < conio.h>
        #include < math.h>
        
        void main()
        {
            float p[5], sum = 0.0, Info[5], I;
            int i, j;
            clrscr();
            do
            {
                sum = 0.0;
                printf("\nEnter the probability of 5 characters:");
                for (i = 0; i < 5; i++)
                {
                    scanf("%f", &p[i]);
                    sum = sum + p[i];
                }
                printf("sum = %f", sum);
            } while (sum != 1.0);
            for (i = 0; i < 5; i++)
            {
                Info[i] = p[i] * (log(1 / p[i]) / log(2));
            }
            I = Info[0] + Info[1] + Info[2] + Info[3] + Info[4];
            printf("\nThe entropy of this probability is: %f", I);
            getch();
        }
        
      
          // Pract 1 b: Occurrences Calculation
            #include < stdio.h>
            #include < conio.h>
            #include < string.h>
            
            void main()
            {
                int i, j, n, count;
                char s[100];
            
                clrscr();
                printf("Enter a string:");
                gets(s);
                n = strlen(s);
            
                for (i = 0; i < n; i++)
                {
                    count = 1;
                    if (s[i])
                    {
                        for (j = i + 1; j < n; j++)
                        {
                            if (s[i] == s[j])
                            {
                                count++;
                                s[j] = '\0';
                            }
                        }
                        printf("'%c' -> %d\n", s[i], count);
                    }
                }
            
                getch();
            }
